# Default

This is the default project generated by Android 23 via:

    android create project \
        --activity AndroidCheatDefault \
        --name AndroidCheatDefault \
        --package com.cirosantilli.android_cheat.default_template \
        --path . \
        --target 1


    android create project \
        --activity AndroidCheatDefault \
        --gradle \
        --gradle-version  \
        --name AndroidCheatDefault \
        --package com.cirosantilli.android_cheat.default_template \
        --path .
